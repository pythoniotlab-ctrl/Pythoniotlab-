from machine import Pin, PWM
import time

# create PWM object on GPIO 18
servo = PWM(Pin(18))

# set PWM frequency to 50Hz
servo.freq(50)

def set_angle(angle):
    # convert angle (0-180) to duty cycle
    duty = int(1000 + (angle / 360) * 8000)
    servo.duty_u16(duty)

while True:
    for angle in range(0, 270, 5):
        set_angle(angle)
        time.sleep(0.05)

    for angle in range(270, -1, -5):
        set_angle(angle)
        time.sleep(0.05)

import random
import time

NUM_PARKING_SPOTS = 5
parking_lot = [False] * NUM_PARKING_SPOTS

def read_sensor(spot):
    return parking_lot[spot]

def update_parking_lot():
    global parking_lot
    for i in range(NUM_PARKING_SPOTS):
        parking_lot[i] = random.choice([True, False])

def display_parking_lot():
    print("Parking Lot Status:")
    for i in range(NUM_PARKING_SPOTS):
        status = "Occupied" if parking_lot[i] else "Vacant"
        print(f"Spot {i+1}: {status}")
    print()

def find_vacant_spot():
    for i in range(NUM_PARKING_SPOTS):
        if not parking_lot[i]:
            return i + 1
    return None

def main():
    print("Welcome to Smart Parking Service!")
    while True:
        update_parking_lot()
        display_parking_lot()
        vacant_spot = find_vacant_spot()

        if vacant_spot:
            print(f"Vacant parking spot available at Spot {vacant_spot}. You can park here!")
        else:
            print("Sorry, no vacant spots available.")

        time.sleep(5)

if __name__ == "__main__":
    main()

from machine import Pin
import dht
import time

# -------------------------
# TM1637 driver (minimal)
# -------------------------
class TM1637:
    def __init__(self, clk, dio):
        self.clk = Pin(clk, Pin.OUT)
        self.dio = Pin(dio, Pin.OUT)
        self.brightness = 0x0f

    def start(self):
        self.dio.value(1)
        self.clk.value(1)
        self.dio.value(0)

    def stop(self):
        self.clk.value(0)
        self.dio.value(0)
        self.clk.value(1)
        self.dio.value(1)

    def write_byte(self, data):
        for _ in range(8):
            self.clk.value(0)
            self.dio.value(data & 1)
            data >>= 1
            self.clk.value(1)
        self.clk.value(0)
        self.clk.value(1)

    def show(self, digits):
        segs = [0x3f,0x06,0x5b,0x4f,0x66,0x6d,0x7d,0x07,
                0x7f,0x6f]
        self.start()
        self.write_byte(0x40)
        self.stop()

        self.start()
        self.write_byte(0xc0)
        for d in digits:
            self.write_byte(segs[d] if d >= 0 else 0x00)
        self.stop()

        self.start()
        self.write_byte(0x88 | self.brightness)
        self.stop()


# -------------------------
# Hardware setup
# -------------------------
dht_sensor = dht.DHT22(Pin(15))   # DHT data pin
display = TM1637(clk=2, dio=3)   # TM1637 pins

# -------------------------
# Main loop
# -------------------------
while True:
    try:
        dht_sensor.measure()
        temp = int(dht_sensor.temperature())

        # Split into digits
        digits = [
            temp // 10,
            temp % 10,
            -1,
            -1
        ]

        display.show(digits)
        print("Temperature:", temp, "°C")

    except Exception as e:
        print("Error:", e)

    time.sleep(2)
